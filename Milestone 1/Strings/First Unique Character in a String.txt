class Solution {
public:
    int firstUniqChar(string s) {
    for(int i=0;i<s.size();i++)
        {
            int temp=0;
            for(int j=0;j<s.size();j++)
            {
                if( s[i] == s[j] && i!=j)
                {
                    temp=1;
                    break;
                }
            }
            
            if(temp==0)
                return i;
        }
        return -1;
        
    }
};
