class Solution {
public:
    bool isPalindrome(string s) {
      
            string t="",pal;
            for(int i=0; i<s.size() ; i++)
            {
                if(isalnum(s[i]))
                {
                    if(isupper(s[i]))
                    {
                        s[i]+=32;
                        t+=s[i];
                    }
                    else
                    {
                        t+=s[i];
                    }
                }
            }
        
            pal=t;
            reverse(pal.begin(),pal.end());

            if(pal==t)
            
                return true;
            
            else
            
                return false;
            
        
    }
};
